SELECT A.STNO,
       C.STNAME,
       C.DEPT,
       A.LECTURECODE,
       A.TIME,
       B.*,
       D.ROOMNUMBER,
       D.BUINDINGNAME,
       D.MAXNO
FROM Q_REGIST_LECTURE A
,Q_LECTURE_INFO B
,Q_STUDENT_INFO C
,Q_CLASSROOM_INFO D 
WHERE 1=1
AND A.LECTURECODE = B.LECCODE(+)
AND A.TIME = B.LESSON(+)
AND A.STNO = C.STUDENT(+)
AND B.LOCATION = D.ROOMCODE(+)
